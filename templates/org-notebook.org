from electrical.converter.buck import design_buck

# Configure your design parameters here:
results = design_buck(
    vin=12,       # Input voltage (V)
    vout=3.3,     # Output voltage (V)
    iout=2.0,     # Output current (A)
    fsw=300e3,    # Switching frequency (Hz)
    vripple=0.01, # Output voltage ripple (V)
    il_ripple_pct=0.30,  # 30% inductor ripple
)

# Print results
for k, v in results.items():
    if isinstance(v, float):
        print(f"{k:25s}: {v:.6e}")
    else:
        print(f"{k:25s}: {v}")

from electrical.utils.plot import figure, show_plot
import matplotlib.pyplot as plt
import numpy as np

# Example: plot inductor current
t = np.linspace(0, 1/300e3, 100)

figure(figsize=(8, 4))
plt.plot(t * 1e6, np.ones_like(t))
plt.xlabel("Time (µs)")
plt.ylabel("Current (A)")
plt.title("Inductor Current Waveform")
plt.grid(True)
show_plot()
